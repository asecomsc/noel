Function miNum(mipar)
Select Case mipar
Case "d"
    elnum = Screen.ActiveControl.Name
    Forms!taller!Text38.SetFocus
    Forms!taller!Text38.SelStart = 0
    SendKeys "{end}" & elnum
Case "b"
    Forms!taller!Text38.SetFocus
    Forms!taller!Text38.SelStart = 0
    SendKeys "{end}{bs}"
Case "e"
    Set rs = CurrentDb.OpenRecordset("SELECT chofer FROM choferes " _
    & "RIGHT JOIN unidades ON choferes.choferID = unidades.choferID " _
    & "WHERE (((unidades.unidadNum)=" & Forms!taller!Text38 & ") AND ((unidades.tipo)=" & Forms!taller!Frame54 & "))")
    Forms!taller!tbChofer = IIf(rs.EOF(), "", rs("chofer"))
End Select
End Function
